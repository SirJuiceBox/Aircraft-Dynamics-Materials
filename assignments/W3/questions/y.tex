\begin{question}
    Consider the following linear ordinary differential equation:
    \begin{equation*}
        a \ddot{y} + b \dot{y} + c y = d u \text{.}
    \end{equation*}
    Write the $A$, $B$, $C$, and $D$ matrices for a state space representation of this system with input $u$ and output $y$.
\end{question}
